Build beautiful, consistent, and accessible user interfaces for your Karrio Apps using our comprehensive component library and design system.

## Overview

Karrio Apps provide access to a rich set of pre-built UI components that follow our design system principles. These components are built with React, TypeScript, and Tailwind CSS, ensuring consistency across all apps in the platform.

## Component Library

### Core Components

#### Button

The `Button` component supports multiple variants, sizes, and states:

```tsx
import { Button } from "@karrio/ui";

export default function MyApp() {
  return (
    <div className="space-y-4">
      {/* Primary button */}
      <Button
        variant="primary"
        size="md"
        onClick={() => console.log("Clicked!")}
      >
        Primary Action
      </Button>

      {/* Secondary button */}
      <Button variant="secondary" size="sm">
        Secondary Action
      </Button>

      {/* Danger button */}
      <Button variant="danger" size="lg" disabled>
        Dangerous Action
      </Button>

      {/* Loading state */}
      <Button variant="primary" loading>
        Loading...
      </Button>

      {/* Icon button */}
      <Button variant="ghost" size="sm" icon="plus">
        Add Item
      </Button>
    </div>
  );
}
```

**Props:**

- `variant`: `primary` | `secondary` | `danger` | `ghost` | `outline`
- `size`: `xs` | `sm` | `md` | `lg` | `xl`
- `loading`: boolean
- `disabled`: boolean
- `icon`: string (icon name)
- `fullWidth`: boolean

#### Input & Form Controls

```tsx
import { Input, Select, Textarea, Checkbox, Radio } from "@karrio/ui";

export default function FormExample() {
  const [formData, setFormData] = useState({
    name: "",
    email: "",
    priority: "medium",
    description: "",
    notifications: false,
    type: "shipment",
  });

  return (
    <form className="space-y-6">
      {/* Text Input */}
      <Input
        label="Full Name"
        placeholder="Enter your full name"
        value={formData.name}
        onChange={(value) => setFormData({ ...formData, name: value })}
        required
        error={!formData.name ? "Name is required" : undefined}
      />

      {/* Email Input */}
      <Input
        type="email"
        label="Email Address"
        placeholder="you@example.com"
        value={formData.email}
        onChange={(value) => setFormData({ ...formData, email: value })}
        leftIcon="mail"
        required
      />

      {/* Select Dropdown */}
      <Select
        label="Priority Level"
        value={formData.priority}
        onChange={(value) => setFormData({ ...formData, priority: value })}
        options={[
          { value: "low", label: "Low Priority" },
          { value: "medium", label: "Medium Priority" },
          { value: "high", label: "High Priority" },
          { value: "urgent", label: "Urgent" },
        ]}
      />

      {/* Textarea */}
      <Textarea
        label="Description"
        placeholder="Describe the task..."
        value={formData.description}
        onChange={(value) => setFormData({ ...formData, description: value })}
        rows={4}
      />

      {/* Checkbox */}
      <Checkbox
        label="Enable notifications"
        checked={formData.notifications}
        onChange={(checked) =>
          setFormData({ ...formData, notifications: checked })
        }
      />

      {/* Radio Group */}
      <div className="space-y-2">
        <label className="text-sm font-medium">Type</label>
        <div className="space-y-2">
          <Radio
            name="type"
            value="shipment"
            label="Shipment"
            checked={formData.type === "shipment"}
            onChange={(value) => setFormData({ ...formData, type: value })}
          />
          <Radio
            name="type"
            value="order"
            label="Order"
            checked={formData.type === "order"}
            onChange={(value) => setFormData({ ...formData, type: value })}
          />
        </div>
      </div>
    </form>
  );
}
```

#### Table

```tsx
import { Table, Badge, Avatar } from "@karrio/ui";

export default function ShipmentTable() {
  const shipments = [
    {
      id: "SH001",
      recipient: "John Doe",
      status: "delivered",
      carrier: "FedEx",
      cost: "$15.99",
      date: "2024-01-15",
    },
    {
      id: "SH002",
      recipient: "Jane Smith",
      status: "in_transit",
      carrier: "UPS",
      cost: "$22.50",
      date: "2024-01-14",
    },
  ];

  const columns = [
    {
      header: "Shipment ID",
      accessor: "id" as const,
      cell: (value: string) => (
        <code className="text-sm bg-gray-100 px-2 py-1 rounded">{value}</code>
      ),
    },
    {
      header: "Recipient",
      accessor: "recipient" as const,
      cell: (value: string) => (
        <div className="flex items-center space-x-2">
          <Avatar name={value} size="sm" />
          <span>{value}</span>
        </div>
      ),
    },
    {
      header: "Status",
      accessor: "status" as const,
      cell: (value: string) => (
        <Badge variant={value === "delivered" ? "success" : "warning"}>
          {value.replace("_", " ")}
        </Badge>
      ),
    },
    {
      header: "Carrier",
      accessor: "carrier" as const,
    },
    {
      header: "Cost",
      accessor: "cost" as const,
      cell: (value: string) => <span className="font-medium">{value}</span>,
    },
    {
      header: "Date",
      accessor: "date" as const,
      cell: (value: string) => (
        <span className="text-gray-500">
          {new Date(value).toLocaleDateString()}
        </span>
      ),
    },
  ];

  return (
    <Table
      data={shipments}
      columns={columns}
      searchable
      sortable
      pagination
      emptyMessage="No shipments found"
      onRowClick={(shipment) => console.log("Clicked:", shipment)}
    />
  );
}
```

### Modal & Overlay Components

#### Modal Dialog

```tsx
import { Modal, ConfirmDialog } from "@karrio/ui";

export default function ModalExample() {
  const [isOpen, setIsOpen] = useState(false);
  const [showConfirm, setShowConfirm] = useState(false);

  const handleDelete = () => {
    setShowConfirm(true);
  };

  const confirmDelete = () => {
    console.log("Item deleted");
    setShowConfirm(false);
  };

  return (
    <>
      <Button onClick={() => setIsOpen(true)}>Open Modal</Button>

      <Modal
        open={isOpen}
        onClose={() => setIsOpen(false)}
        title="Create New Shipment"
        size="lg"
      >
        <Modal.Content>
          <form className="space-y-4">
            <Input label="Recipient Name" placeholder="Enter recipient name" />
            <Input
              label="Tracking Number"
              placeholder="Enter tracking number"
            />
            <Select
              label="Carrier"
              options={[
                { value: "fedex", label: "FedEx" },
                { value: "ups", label: "UPS" },
                { value: "dhl", label: "DHL" },
              ]}
            />
          </form>
        </Modal.Content>
        <Modal.Footer>
          <Button variant="outline" onClick={() => setIsOpen(false)}>
            Cancel
          </Button>
          <Button variant="primary" onClick={() => setIsOpen(false)}>
            Create Shipment
          </Button>
        </Modal.Footer>
      </Modal>

      {/* Confirmation Dialog */}
      <Button variant="danger" onClick={handleDelete}>
        Delete Item
      </Button>

      <ConfirmDialog
        open={showConfirm}
        onClose={() => setShowConfirm(false)}
        onConfirm={confirmDelete}
        title="Confirm Deletion"
        message="Are you sure you want to delete this item? This action cannot be undone."
        confirmText="Delete"
        cancelText="Cancel"
        variant="danger"
      />
    </>
  );
}
```

## Accessibility

### Built-in Accessibility Features

All Karrio UI components include accessibility features:

- **Keyboard Navigation**: Full keyboard support with proper focus management
- **Screen Reader Support**: ARIA labels and descriptions
- **High Contrast**: Meets WCAG contrast requirements
- **Focus Management**: Visible focus indicators and logical tab order

```tsx
// Accessibility example
<form>
  <Input
    label="Email Address"
    id="email"
    type="email"
    required
    aria-describedby="email-error"
    error="Please enter a valid email address"
  />
  <div id="email-error" role="alert" aria-live="polite">
    {/* Error message will be announced by screen readers */}
  </div>
</form>
```

### Custom Accessibility

```tsx
import { useAnnouncer, useFocusManager } from "@karrio/ui";

export default function AccessibleComponent() {
  const { announce } = useAnnouncer();
  const { focusFirst, focusLast } = useFocusManager();

  const handleAction = () => {
    // Announce status to screen readers
    announce("Task completed successfully", "polite");

    // Manage focus programmatically
    focusFirst('[data-focusable="true"]');
  };

  return (
    <div>
      <Button onClick={handleAction}>Complete Task</Button>
      <div data-focusable="true" tabIndex={0}>
        Focusable element
      </div>
    </div>
  );
}
```

### State Management

```tsx
// ✅ Good: Use controlled components with proper state management
export default function ShippingForm() {
  const [formState, setFormState] = useState({
    recipient: "",
    address: "",
```

### Performance Optimization

```tsx
import { memo, useMemo, useCallback } from "react";

// ✅ Good: Memoize expensive operations
const ShipmentList = memo(({ shipments, onUpdate }) => {
  const sortedShipments = useMemo(() => {
    return shipments.sort(
      (a, b) =>
        new Date(b.created_at).getTime() - new Date(a.created_at).getTime(),
    );
  }, [shipments]);

  const handleUpdate = useCallback(
    (id, updates) => {
      onUpdate(id, updates);
    },
    [onUpdate],
  );

  return (
    <div>
      {sortedShipments.map((shipment) => (
        <ShipmentCard
          key={shipment.id}
          shipment={shipment}
          onUpdate={handleUpdate}
        />
      ))}
    </div>
  );
});
```
